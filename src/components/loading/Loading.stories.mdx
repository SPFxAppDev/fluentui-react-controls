# Loading

Accordion element that expands when the user clicks on the header. If the icon is present it is rotated 90deg when item is expanded.

## Arguments

<ArgTypes of={Loading} />

export const BasicTemplate = (args) => <div></div>;

<Canvas>
  <Story name="Basic">
  {(args) => {
      const [showLoading, setShowLoading] = useState(false);
      return (
        <div>
        <PrimaryButton
        onClick={() => {
            setShowLoading(true);

            var timer = setTimeout(() => {
            setShowLoading(false);
            timer = void 0
            }, 3000);
        }}
        >Show Loading</PrimaryButton>
        <Loading hidden={!showLoading} />
        </div>
      );

}}

  </Story>
</Canvas>
